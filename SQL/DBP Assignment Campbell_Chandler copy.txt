—Prompt A: Creating the database and tables for the database

# Set up the database

CREATE DATABASE library;

# Library database schema

/* Creating an index on all of the table columns to improve query performance *\

# Schema for client table

CREATE TABLE Client
(
ClientId INT PRIMARY KEY,
ClientFirstName VARCHAR(12),
ClientLastName VARCHAR(12),
ClientDoB YEAR,
Occupation VARCHAR(30),
INDEX (ClientId, ClientFirstName, ClientLastName, ClientDOB)
);


# Schema for author table

CREATE TABLE Author
(
AuthorId INT PRIMARY KEY,
AuthorFirstName VARCHAR(12),
AuthorLastName VARCHAR(12),
AuthorNationality VARCHAR(15),
INDEX (AuthorId, AuthorFirstName, AuthorLastName, AuthorNationality)
);


# Schema for book table

CREATE TABLE Book
(
BookId INT PRIMARY KEY,
BookTitle VARCHAR(40),
BookAuthor INT NOT NULL,
Genre VARCHAR(12),
FOREIGN KEY (BookAuthor)
	REFERENCES Author(AuthorId),
INDEX (BookId, BookTitle, BookAuthor, Genre)
);


# Schema for borrower table

CREATE TABLE Borrower
(
BorrowId INT PRIMARY KEY,
ClientId INT NOT NULL,
BookId INT NOT NULL,
BorrowDate DATE,
FOREIGN KEY (ClientId)
	REFERENCES Client(ClientId),
FOREIGN KEY (BookId)
	REFERENCES Book(BookId),
INDEX (BorrowId, ClientId, BookId, BorrowDate)
);

# Query 9: Average number of borrowed books by job title

/* This one was difficult to figure out and uses aliases to simplify the complex query. To calculate an average here, divide the number of books checked out by the different occupations. *\

WITH ClientCount AS (SELECT COUNT(Client.ClientId) AS PeopleWJob, Client.Occupation
FROM Client
GROUP BY Client.Occupation), 
BorrowerCount AS (SELECT COUNT(Borrower.BorrowId) AS CheckedOut, Client.Occupation
FROM Borrower
INNER JOIN Book
ON Borrower.BookId = Book.BookId
INNER JOIN Client
ON Borrower.ClientId = Client.ClientId
GROUP BY Client.Occupation
ORDER BY CheckedOut DESC)
SELECT ClientCount.Occupation, (BorrowerCount.CheckedOut / ClientCount.PeopleWJob) AS AverageBorrows
FROM ClientCount
JOIN BorrowerCount
ON ClientCount.Occupation = BorrowerCount.Occupation
ORDER BY AverageBorrows DESC;

# Query 10: Create a VIEW and display the titles that were borrowed by at least 20% of clients

/* Views are used to simplify the querying process and for database security. To create this specific view, we need to use a nested subquery, which is then used to run the outer query *\

CREATE VIEW MostPopularBooks AS 
SELECT Book.BookTitle, COUNT(Client.ClientId) AS CheckOuts
FROM Borrower
JOIN Book
ON Book.BookId = Borrower.BookId
JOIN Client
ON Borrower.ClientId = Client.ClientId
GROUP BY Book.BookTitle
HAVING CheckOuts >= 
(SELECT (COUNT(Client.ClientId) * 0.2)
FROM Client);

# Query 11: The top month of borrows in 2017

/* This query uses aggregate data so grouping by the month of the borrow date is necessary *\

FROM Borrower

# Query 12: Average number of borrows by age

/* Have to use date arithmetic, inner joins, and grouping again *\

SELECT AVG(numBorrowsByAge.CheckedOut)
FROM
(SELECT COUNT(Borrower.BorrowId) AS CheckedOut, (2023 - Client.ClientDoB) AS Age
FROM Borrower
INNER JOIN Book
ON Borrower.BookId = Book.BookId
INNER JOIN Client
ON Borrower.ClientId = Client.ClientId
GROUP BY Age
ORDER BY CheckedOut DESC) numBorrowsByAge;

# Query 13: The oldest and the youngest clients of the library

/* We need to find the max and min of age, first converting date of birth to age using date arithmetic *\

SELECT (2023 - Client.ClientDoB) AS Age, Client.ClientFirstName AS FirstName, Client.ClientLastName AS LastName
FROM Client
HAVING Age = 
(SELECT MAX(2023 - Client.ClientDoB)
FROM Client) OR Age =
(SELECT MIN(2023 - Client.ClientDoB)
FROM Client)
ORDER BY Age;
